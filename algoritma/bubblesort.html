Algoritma BubbleSort


Algoritma pengurutan dimana membandingkan data n[min] > n[current], lalu set dana n[current] ke data terkecil dimana data ini akan di swap dengan data pertama.


Implementasi

/**
 * Algorithm BubleSort
 * @param array
 * @timeComplexity O(n2)
 * @spaceComplexity O(n2)
 */
public static void bubbleSort(int[] array) {
    int n = array.length;
    for(int i = 0; i < n-1; i++) {
        for(int j = 0; j < n-1; j++) {
            if(array[j] > array[j+1]) {
                int swap = array[j];
                array[j] = array[j+1];
                array[j+1] = swap;
            }
        }
    }
}
