Algoritma MergeSort


Algoritma MergeSort berprinsip algoritma yang membagi lalu menggabungkan.

Salin
                
import java.util.*;

public class MergeSort {
    public static void main(String... args) {
        int[] arr1 = {
            1, 3, 5, 10, 9, 7, 8, 2, 20, 15, 18, 13, 19
        };
        algo(arr1);
        System.out.println(Arrays.toString(arr1));
    }

    public static void algo(int[] nums) {
        int length = nums.length;
        if(length <= 1) {
            return;
        }
        int middle = (length/2);
        int[] left = new int[middle];
        int[] right = new int[length - middle];
        int rightIndex = 0;
        for(int i = 0; i < length; i++) {
            if(i < middle) {
                left[i] = nums[i];
            }
            else {
                right[rightIndex++] = nums[i];
            }
        }
        algo(left);
        algo(right);
        merge(left, right, nums);
    }

    public static void merge(int[] arrLeft, int[] arrRight, int[] arrOri) {
        int leftIndex = 0,
            rightIndex = 0,
            arrIndex = 0,
            leftSize = arrLeft.length,
            rightSize = arrRight.length;
        
        while(leftIndex < leftSize || rightIndex < rightSize) {
            if(leftIndex < leftSize && rightIndex < rightSize) {
                if(arrLeft[leftIndex] < arrRight[rightIndex]) { arrOri[arrIndex++] = arrLeft[leftIndex++]; }
                else { arrOri[arrIndex++] = arrRight[rightIndex++]; }
            }
            else {
                if(leftIndex < leftSize) { arrOri[arrIndex++] = arrLeft[leftIndex++]; }
                else if(rightIndex < rightSize) { arrOri[arrIndex++] = arrRight[rightIndex++]; }
            }   
        }
    }
}
